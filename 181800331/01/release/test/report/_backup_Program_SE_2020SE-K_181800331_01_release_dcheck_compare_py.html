15def get_jac_idx(s1, s2): 

16 """ Get Jaccard index. 

17 

18 :param s1: word set 1 

19 :param s2: word set 2 

20 :return: whether both of the sets are empty and Jaccard index 

21 """ 

22 

23 if len(s1) == 0 and len(s2) == 0: 

24 return True, 1.0 

25 

26 inter = s1 & s2 

27 union = s1 | s2 

28 return False, len(inter) / len(union)

31def get_tf(texts, inter): 

32 """ Get term frequency. 

33 

34 :param texts: lists of words 

35 :param inter: word domain 

36 :return: lists of term frequency 

37 """ 

38 

39 dictionary = corpora.Dictionary(texts) 

40 total = [len(text) for text in texts] 

41 

42 corpus = [dictionary.doc2bow(text) for text in texts] 

43 tf = np.zeros((len(texts), len(dictionary))) 

44 for i, vec in enumerate(corpus): 

45 for word in vec: 

46 if dictionary[word[0]] in inter: 

47 tf[i][word[0]] = word[1] / total[i] 

48 

49 return tf

52def get_cos_sim(s1, s2, texts): 

53 """ Get cosine similarity. 

54 

55 :param s1: word set 1 

56 :param s2: word set 2 

57 :param texts: lists of words 

58 :return: whether both of the sets are not empty and cosine similarity 

59 """ 

60 

61 inter = s1 & s2 

62 if len(inter) == 0: 

63 return False, 0.0 

64 

65 tf = get_tf(texts, inter) 

66 

67 cs = np.dot(tf[0], tf[1]) / (np.linalg.norm(tf[0]) * np.linalg.norm(tf[1])) 

68 return True, cs

71def get_sim(texts): 

72 """ Get similarity. 

73 

74 :param texts: lists of words 

75 :return: similarity of the two sets 

76 """ 

77 

78 ori_dic, smp_dic = pc.word2dic(texts) 

79 ori_set = set(ori_dic.token2id.keys()) 

80 smp_set = set(smp_dic.token2id.keys()) 

81 

82 f1, jc = get_jac_idx(ori_set, smp_set) 

83 f2, cs = get_cos_sim(ori_set, smp_set, texts) 

84 

85 if f1: 

86 return 1.0 

87 elif not f2: 

88 return 0.0 

89 else: 

90 return jc * cs
